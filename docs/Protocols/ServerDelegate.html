ServerDelegate

public protocol ServerDelegate: class

The protocol defining the delegate for the HTTPServer and the FastCGIServer classes. The delegate’s handle function is invoked when new requests arrive at the server for processing.

Usage Example:

func handle(request: ServerRequest, response: ServerResponse) {

    let parsedURL = URLParser(url: request.url, isConnect: false)

    if parsedURL.path == "/hello" {
       let payload = "Hello, world!"
       response.headers["Content-Type"] = ["text/plain"]
       response.headers["Content-Length"] = [String(payload.characters.count)]
       response.statusCode = .OK
       do {
           try response.write(from: payload)
           try response.end()
       }
       catch {
           print("Failed to write the response. Error: \(error)")
       }
}
